Star Patterns (C#)

Code: 1
          
using System;

class Program {
    static void Main() {
        for (int i = 1; i <= 5; i++) 
        {
            for (int j = 1; j <= 5; j++) 
            {
                Console.Write("* ");
            }
            Console.WriteLine();
        }
    }
}
Output
* * * * *
* * * * *
* * * * *
* * * * *
