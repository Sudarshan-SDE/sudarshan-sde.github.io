Array Programs (C#)

Code: 1 (Reverse the array -> Using Reverse Method)
            
using System;

class Program {
    static void Main() {
        int[] arr = {10, 20, 30, 40, 50};
          
        // Reverse the array
        Array.Reverse(arr);
          
        // Print the reversed array
        foreach (int i in arr)
        {
          Console.Write(i + " ");
        }
    }
}
Output
50 40 30 20 10
Code: 2 (Reverse the array -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        int[] arr = { 10, 20, 30, 40, 50 };
        
        for (int i = arr.Length - 1; i >= 0; i--)
        {
          Console.Write(arr[i] + " ");
        }
        Console.WriteLine();
    }
}
Output
50 40 30 20 10

    }
}
Output

Code: 4 (Find Max Element -> Using Foreach Loop)
            
using System;

class Program {
    static void Main() {
        int[] arr = { 52, 36, 96, 78 };

        int max = arr[0];

        foreach (int i in arr)
        {
            if (i > max)
            {
                max = i;
            }
        }
        Console.WriteLine(max);
    }
}
Output
96
