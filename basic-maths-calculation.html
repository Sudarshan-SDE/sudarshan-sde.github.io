Code 1: (Find Even Odd Number)
            
using System;

class Program {
  static void Main() {

    Console.Write("Enter the Number: ");
    int num = Convert.ToInt32(Console.ReadLine());
    
    if (num % 2 == 0)
    {
      Console.WriteLine($"{num} is Even!");
    }
    else
    {
      Console.WriteLine($"{num} is odd!");
    }

  }
}
Output
Enter the Number: 8
8 is Even!
Code 2: (Find Largest of Three Numbers)
            
using System;

class Program {
  static void Main() {

    int num1, num2, num3;

    Console.Write("Enter the Number 1: ");
    num1 = Convert.ToInt32(Console.ReadLine());

    Console.Write("Enter the Number 2: ");
    num2 = Convert.ToInt32(Console.ReadLine());

    Console.Write("Enter the Number 3: ");
    num3 = Convert.ToInt32(Console.ReadLine());

    if (num1 > num2 && num1 > num3)
    {
        Console.WriteLine($"{num1} is Largest");
    }
    else if (num2 > num1 && num2 > num3)
    {
        Console.WriteLine($"{num2} is Largest");
    }
    else
    {
        Console.WriteLine($"{num3} is Largest");
    }
    
  }
}
Output
Enter the Number 1: 5
Enter the Number 2: 10
Enter the Number 3: 3
10 is Largest
Code 3: (Swap Number Using 2 Variables)
            
using System;

class Program {
  static void Main() {

    int num1, num2;

    Console.Write("Enter the Number 1: ");
    num1 = Convert.ToInt32(Console.ReadLine());

    Console.Write("Enter the Number 2: ");
    num2 = Convert.ToInt32(Console.ReadLine());

    Console.WriteLine($"Before swaping: num1 is -> {num1} and num2 is -> {num2}");

    num1 = num1 + num2;
    num2 = num1 - num2;
    num1 = num1 - num2;

    Console.WriteLine();
    Console.WriteLine($"After swaping: num1 is -> {num1} and num2 is -> {num2}");
        
  }
}
Output
 
Enter the Number 1: 5
Enter the Number 2: 10  
Before swaping: num1 is -> 5 and num2 is -> 10
After swaping: num1 is -> 10 and num2 is -> 5
Code 4: (Swap Number Using 3 Variables)
            
using System;

class Program {
  static void Main() {

    int num1, num2, temp;

    Console.Write("Enter the Number 1: ");
    num1 = Convert.ToInt32(Console.ReadLine());

    Console.Write("Enter the Number 2: ");
    num2 = Convert.ToInt32(Console.ReadLine());

    Console.WriteLine($"Before swaping: num1 is -> {num1} and num2 is -> {num2}");

    temp = num1;
    num1 = num2;
    num2 = temp;

    Console.WriteLine($"After swaping: num1 is -> {num1} and num2 is -> {num2}");
        
  }
}
Output
 
Enter the Number 1: 5
Enter the Number 2: 10  
Before swaping: num1 is -> 5 and num2 is -> 10
After swaping: num1 is -> 10 and num2 is -> 5

Code 6: (Reverse the Number)
            
using System;

class Program {
  static void Main() {

    int num, reversed = 0, rem;

    Console.Write("Enter the Number: ");
    num = Convert.ToInt32(Console.ReadLine());

    while (num != 0)
    {
        rem = num % 10;
        reversed = reversed * 10 + rem;
        num = num / 10;
    }
    Console.Write("Reversed Number: " + reversed);
        
  }
}
Output
 
Enter the Number: 1234        
Reversed Number: 4321
Code 7: (Sum of Numbers)
            
using System;

class Program {
  static void Main() {

    Console.Write("Enter the Number: ");
