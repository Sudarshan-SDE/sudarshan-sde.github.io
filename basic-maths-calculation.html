String Programs (C#)

Code 1: (Reverse the Predefined string -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        string words = "Sudarshan Sharma";
        string reversed = "";

        for (int i = words.Length - 1; i >= 0; i--)
        {
            reversed = reversed + words[i];
        }
        Console.WriteLine(reversed);
    }
}
Output
Reversed String: amrahS nahsraduS
