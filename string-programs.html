String Programs (C#)

Code: 1 (Reverse the string -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        string words = "Sudarshan Sharma";
        string reversed = "";

        for (int i = words.Length - 1; i >= 0; i--)
        {
            reversed = reversed + words[i];
        }
        Console.WriteLine(reversed);
    }
}
Output
amrahS nahsraduS
