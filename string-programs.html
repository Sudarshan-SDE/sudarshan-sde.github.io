String Programs (C#)

Code 1: (Reverse the Predefined string -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        string words = "Sudarshan Sharma";
        string reversed = "";

        for (int i = words.Length - 1; i >= 0; i--)
        {
            reversed = reversed + words[i];
        }
        Console.WriteLine(reversed);
    }
}
Output
Reversed String: amrahS nahsraduS
Code 2: (Reverse the User input string -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        Console.Write("Enter the string: ");
        string words = Console.ReadLine();
        string reversed = "";

        for (int i = words.Length - 1; i >= 0; i--)
        {
            reversed = reversed + words[i];
        }
        Console.WriteLine("Reversed String: "+reversed);
    }
}
Output
Enter the string: Sudarshan Sharma
Reversed String: amrahS nahsraduS
Code 3: (Reverse the Predefined string (word by word) -> Using For Loop)
            
using System;

class Program {
    static void Main() {
        string input_String = "Sudarshan Sharma";
        string[] words = input_String.Split(' ');
        string result = "";

        foreach (string word in words)
        {
            string reversed = "";

            for (int i = word.Length - 1; i >= 0; i--)
            {
                reversed = reversed + word[i];
            }
            result = result + reversed + " ";
        }
        Console.WriteLine(result);
    }
}
Output
nahsraduS amrahS

Code 5: (Remove space from the string -> Using Method)
            
using System;

class Program {
    static void Main() {
        Console.Write("Enter the String: ");
        string input_String = Console.ReadLine();
        string result = input_String.Replace(" ", "");
        Console.WriteLine("Remove Space String: " + result);
    }
}
Output
Enter the String: Sudarshan Sharma
Remove Space String: nahsraduSamrahS
